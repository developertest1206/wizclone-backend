CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Copy limit check karva maate — backend call kare before every copy
CREATE OR REPLACE FUNCTION check_copy_allowed(p_workspace_id UUID)
RETURNS JSONB AS $$
DECLARE
  v_plan     plans%ROWTYPE;
  v_sub      workspace_subscriptions%ROWTYPE;
  v_usage    usage_metrics%ROWTYPE;
BEGIN
  SELECT p.* INTO v_plan
    FROM plans p
    JOIN workspace_subscriptions ws ON ws.plan_id = p.id
    WHERE ws.workspace_id = p_workspace_id AND ws.is_active = true
    LIMIT 1;

  SELECT * INTO v_usage
    FROM usage_metrics
    WHERE workspace_id = p_workspace_id
      AND billing_cycle_start <= now()::DATE
      AND billing_cycle_end   >= now()::DATE
    LIMIT 1;

  -- NULL = unlimited (BUSINESS plan)
  IF v_plan.max_copies_per_month IS NULL THEN
    RETURN jsonb_build_object('allowed', true, 'reason', 'unlimited');
  END IF;

  IF COALESCE(v_usage.copies_used, 0) >= v_plan.max_copies_per_month THEN
    RETURN jsonb_build_object(
      'allowed',    false,
      'reason',     'monthly_limit_reached',
      'used',       COALESCE(v_usage.copies_used, 0),
      'limit',      v_plan.max_copies_per_month,
      'plan',       v_plan.plan_name
    );
  END IF;

  RETURN jsonb_build_object(
    'allowed',    true,
    'remaining',  v_plan.max_copies_per_month - COALESCE(v_usage.copies_used, 0)
  );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Usage atomically increment karvaa — thread-safe
CREATE OR REPLACE FUNCTION increment_copy_usage(
  p_workspace_id UUID,
  p_ai_used      BOOLEAN DEFAULT false
) RETURNS VOID AS $$
BEGIN
  UPDATE usage_metrics SET
    copies_used      = copies_used + 1,
    ai_match_calls   = ai_match_calls + CASE WHEN p_ai_used THEN 1 ELSE 0 END,
    last_copy_at     = now(),
    updated_at       = now()
  WHERE workspace_id = p_workspace_id
    AND billing_cycle_start <= now()::DATE
    AND billing_cycle_end   >= now()::DATE;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
